## 题目
给定一个链表，其中奇数位是升序的，偶数位是降序的，实现链表的排序

时复杂度为O(n),空间复杂度为O（1）

&nbsp;
## 题解
### 思路
1、将原始链表按奇偶分开（leetcode原题：奇偶链表）  
2、偶链表是降序的，所以将其逆序（leetcode原题：翻转链表）  
3、合并两个有序链表（leetcode原题：合并两个有序链表）

### 自己写
```
class Solution:
  def sortOddEvenList(head:ListNode)->ListNode:
    if not head:return
    
    ### 第一步：奇偶链表
    odd,even=head,head.next
    oddHead,evenHead=odd,even
    while even and even.next:
      odd.next=even.next
      odd=odd.next
      even.next=odd.next
      even=even.next
    
    odd.next=None  /// 断开，为了翻转偶数链表
    ### 第二步：翻转偶数链表
    pre=None
    even=evenHead
    while even:
      tmp=even.next  ///临时存储下一个节点
      even.next=pre  ///even指针指向pre
      pre=even        /// 更换迭代器pre为even
      even=tmp        /// 更换迭代器even为tmp
   
   evenHead=pre  ///更换偶数链表头节点
   
   ### 第三步：合并
   preHead=ListNode(-1)   ///哨兵节点，便于排序后返回真正的头节点
   pre,odd,even=preHead,oddHead,evenHead
   while odd and even:
    if odd.val<even.val:
      pre.next=odd
      odd=odd.next
    else:
      pre.next=even
      even=even.next
    pre=pre.next ///更换迭代器pre为下一个（odd或even中的较小值）
    
    pre.next=odd if odd else even  ///如果剩余奇数不为空，preHead指向剩余奇数，否则指向偶数
    
  return preHead.next
```

package main

import "fmt"

//字节面试题：给定奇偶链表，排序
//如1->8->3->6->5->4->7->2
//奇数位是递增，偶数位递减，排序

//定义链表
type ListNode struct {
	Val  int
	Next *ListNode
}

func sortEvenAndOddLinkedList(head *ListNode) *ListNode {
	if head == nil || head.Next == nil {
		return head
	}
	//翻转奇偶链表
	Odd := head       //奇数节点的起点
	Even := head.Next //偶数节点的起点
	oddHead := Odd
	evenHead := Even
	for Even != nil && Even.Next != nil {
		Odd.Next = Even.Next
		Odd = Odd.Next
		Even.Next = Odd.Next
		Even = Even.Next
	}
	Odd.Next = nil
	//将偶数的链表进行翻转
	p := reverseLinkedList(evenHead)
	//合并两个排序的链表
	res := mergeSortEvenAndOdd(oddHead, p)
	return res
}

//翻转链表
func reverseLinkedList(p *ListNode) *ListNode {
	var pre *ListNode
	for p != nil {
		tmp := p.Next
		p.Next = pre
		pre = p
		p = tmp
	}
	return pre
}

//合并两个有序链表
func mergeSortEvenAndOdd(p *ListNode, q *ListNode) *ListNode {
	dummy := &ListNode{-1, nil}
	cur := dummy
	for p != nil && q != nil {
		if p.Val < q.Val {
			cur.Next = p
			cur = cur.Next
			p = p.Next
		} else {
			cur.Next = q
			cur = cur.Next
			q = q.Next
		}
	}
	if p != nil {
		cur.Next = p
	}
	if q != nil {
		cur.Next = q
	}
	return dummy.Next
}

//创建链表
func createLinkedList(nums []int) *ListNode {
	dummy := &ListNode{-1, nil}
	head := &ListNode{nums[0], nil}
	dummy.Next = head
	for i := 1; i < len(nums); i++ {
		head.Next = &ListNode{nums[i], nil}
		head = head.Next
	}
	return dummy.Next
}

//打印链表
func printLinkedList(head *ListNode) {
	fmt.Println("------------------------------------------------------------")
	for head != nil {
		fmt.Printf("%d\t", head.Val)
		head = head.Next
	}
	fmt.Println()
}

func main() {
	nums := []int{1, 8, 3, 6, 5, 4, 7, 2}
	head := createLinkedList(nums)
	printLinkedList(head)
	head2 := sortEvenAndOddLinkedList(head)
	printLinkedList(head2)
}
